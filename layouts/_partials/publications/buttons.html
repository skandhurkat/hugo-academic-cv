{{/* Renders the action buttons for a publication: PDF, ArXiv, DOI, BibTeX. Usage: {{ partial "publications/buttons.html" . }} (pass the publication page as context) PDF resolution order: 1. Full URL (http/https/protocol-relative) — used as-is 2. Root-relative path (/papers/foo.pdf) — passed through relURL 3. Page bundle resource (filename only) — resolved via .Resources.GetMatch This supports page bundles: put the PDF next to the index.md and reference it by filename in front matter. */}} {{ $page := . }}
{{- with $page.Params.pdf }} {{ $url := . }} {{ $finalURL := "" }} {{ if or (hasPrefix $url "http://") (hasPrefix $url "https://") (hasPrefix $url "//") }} {{ $finalURL = $url }} {{ else if hasPrefix $url "/" }} {{ $finalURL = $url | relURL }} {{ else }} {{ with $page.Resources.GetMatch $url }} {{ $finalURL = .RelPermalink }} {{ else }} {{/* Warn at build time rather than silently producing a broken button */}} {{ warnf "publications/buttons.html: could not find PDF resource %q in page %s" $url $page.RelPermalink }} {{ end }} {{ end }} {{ with $finalURL }} PDF {{ end }} {{- end }} {{- with $page.Params.arxiv }} arXiv {{- end }} {{- with $page.Params.doi }} DOI {{- end }} {{/* --- Derive BibTeX entry type from publication_types --- */}} {{ $pubtype := "misc" }} {{ with $page.Params.publication_types }} {{ $first := index . 0 }} {{ if eq $first "conference" }}{{ $pubtype = "inproceedings" }} {{ else if eq $first "journal" }}{{ $pubtype = "article" }} {{ else if eq $first "preprint" }}{{ $pubtype = "misc" }} {{ else if eq $first "thesis" }}{{ $pubtype = "phdthesis" }} {{ else if eq $first "book" }}{{ $pubtype = "book" }} {{ else if eq $first "bookchapter" }}{{ $pubtype = "incollection" }} {{ end }} {{ end }} {{/* --- Build cite key: FirstAuthorLastname + Year --- */}} {{ $year := $page.Date.Format "2006" }} {{ $citekey := "Unknown" }} {{ with $page.Params.authors }} {{ $firstAuthor := index . 0 }} {{/* Take the last word of the first author's name as the last name */}} {{ $parts := split $firstAuthor " " }} {{ $lastname := index $parts (sub (len $parts) 1) }} {{ $citekey = printf "%s%s" $lastname $year }} {{ end }} {{/* --- Assemble author string (BibTeX uses " and " as separator) --- */}} {{ $authorStr := "" }} {{ with $page.Params.authors }} {{ $authorStr = delimit . " and " }} {{ end }} {{/* --- Build the BibTeX entry as a string --- */}} {{ $bib := printf "@%s{%s,\n" $pubtype $citekey }} {{ $bib = printf "%s author = {%s},\n" $bib $authorStr }} {{ $bib = printf "%s title = {%s},\n" $bib $page.Title }} {{ $bib = printf "%s year = {%s},\n" $bib $year }} {{ with $page.Params.venue }}{{ $bib = printf "%s booktitle = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.booktitle }}{{ $bib = printf "%s booktitle = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.publisher }}{{ $bib = printf "%s publisher = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.volume }}{{ $bib = printf "%s volume = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.number }}{{ $bib = printf "%s number = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.pages }}{{ $bib = printf "%s pages = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.doi }}{{ $bib = printf "%s doi = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.arxiv }}{{ $bib = printf "%s eprint = {%s},\n" $bib . }}{{ end }} {{ $bib = printf "%s}" $bib }} {{/* BibTeX button — triggers a Bootstrap modal defined in bibtex_modal.html. The modal is included once in the page by pub_card.html or single.html; the button here just references it by ID. data-bibtex is written by bibtex_modal.html, not here. */}}
