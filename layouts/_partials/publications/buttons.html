{{/* Renders the action buttons for a publication: PDF, ArXiv, DOI, BibTeX. Usage: {{ partial "publications/buttons.html" . }} (pass the publication page as context) PDF resolution order: 1. Full URL (http/https/protocol-relative) — used as-is 2. Root-relative path (/papers/foo.pdf) — passed through relURL 3. Page bundle resource (filename only) — resolved via .Resources.GetMatch This supports page bundles: put the PDF next to the index.md and reference it by filename in front matter. */}} {{ $page := . }}
{{- with $page.Params.pdf }} {{ $url := . }} {{ $finalURL := "" }} {{ if or (hasPrefix $url "http://") (hasPrefix $url "https://") (hasPrefix $url "//") }} {{ $finalURL = $url }} {{ else if hasPrefix $url "/" }} {{ $finalURL = $url | relURL }} {{ else }} {{ with $page.Resources.GetMatch $url }} {{ $finalURL = .RelPermalink }} {{ else }} {{/* Warn at build time rather than silently producing a broken button */}} {{ warnf "publications/buttons.html: could not find PDF resource %q in page %s" $url $page.RelPermalink }} {{ end }} {{ end }} {{ with $finalURL }} PDF {{ end }} {{- end }} {{- with $page.Params.arxiv }} arXiv {{- end }} {{- with $page.Params.doi }} DOI {{- end }} {{/* BibTeX button — triggers a Bootstrap modal defined in bibtex_modal.html. The modal is included once in the page by pub_card.html or single.html; the button here just references it by ID. data-bibtex is written by bibtex_modal.html, not here. */}}
